A node style accessor, that fetches the given id, calling the given callback on completion, with either an error value, or the model.

 
  @find: (id, callback) ->
    model = new @()
    model.set(model.idAttribute, id)
    success = (m) ->
      callback(null, m)
    error = (m) ->
      callback(m)
    model.fetch {success, error}
